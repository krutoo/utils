# `useOutsideClick`

export const meta = {
  category: 'React',
  title: 'useOutsideClick',
};

Hook for handle click outside some element.

```tsx
import { useRef } from 'react';
import { useOutsideClick } from '@krutoo/utils/react';

interface TooltipProps {
  onDismiss?: VoidFunction;
}

function Tooltip({ onDismiss }: TooltipProps) {
  const ref = useRef<HTMLDivElement>(null);

  useOutsideClick(ref, event => {
    onDismiss?.();
  });

  return <div ref={ref}>My tooltip</div>;
}
```
